CREATE TABLE public.brasil (
    gid integer  NOT NULL,
    municipio character varying(254),
    cidade character varying(254),
    codmun bigint,
    codibge bigint,
    uf character varying(254),
    ordem bigint,
    geom public.geometry(Point,4326),
